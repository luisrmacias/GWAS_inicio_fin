##venimos de guión conversión vcf en el que se establece carpeta de trabajo
##cd ~/lmacias/analisis_asociacion/adultos_mega/calidad/cluster3
cd imputacion
mkdir faseados
car_referencias=~/CANDELA/haplotipos/indigenas/MEGA/universales_indigenas
estudio=adultos
for chr in {1..6}
do
bcftools query -f '%CHROM\t%POS\t%REF\t%ALT\n' vcf/adultos.$chr.vcf | grep -vw -e A -e C -e T -e G | cut -f 1,2 > temp.txt
bcftools view -T ^temp.txt vcf/adultos.$chr.vcf -oz -O temp.vcf.gz
java -jar ~/scripts/beagleyut/conform-gt.24May16.cee.jar ref=$car_referencias/nat_uni_extendido.$chr.vcf.gz gt=temp.vcf.gz chrom=$chr match=ID out=mod.$estudio.$chr
nohup java -jar ~/scripts/beagleyut/beagle.26May18.835.jar ref=$car_referencias/nat_uni_extendido.$chr.vcf.gz gt=mod.$estudio.$chr.vcf.gz map=~/CANDELA/haplotipos/ref_beagle/listas/mapa_beagle/plink.chr$chr.GRCh37.map iterations=5 nthreads=20 window=100 overlap=20 out=faseados/$estudio.chr$chr
done
